# word	        the text that will be inserted, mandatory
# abbr	        abbreviation of "word"; when not empty it is used in
# 	        the menu instead of "word"
# menu	        extra text for the popup menu, displayed after "word"
# 	        or "abbr"
## info	        more information about the item, can be displayed in a
# 	        preview window
# kind	        single letter indicating the type of completion
# icase	        when non-zero case is to be ignored when comparing
# 	        items to be equal; when omitted zero is used, thus
# 	        items that only differ in case are added
# dup	        when non-zero this match will be added even when an
# 	        item with the same word is already present.
# 

# HTML
word  ~> =html#htags#paired('html','') 
abbr  ~> thtml
menu  ~> <html>
--
# <!DOCTYPE html>
word  ~> =html#htags#doctype_html() 
abbr  ~> dh
menu  ~> <!DOCTYPE html>
--
# BODY
word  ~> =html#htags#paired('body','') 
abbr  ~> tbody
menu  ~> <html>
--
# FORM
word  ~> =html#htags#paired('form','action="mza" method="post"') 
abbr  ~> tform
menu  ~> <html>
--
# INPUT
word  ~> =html#htags#single('input','type="mzatext" name="mxa"')`xa 
abbr  ~> tinput
menu  ~> <html>
--
# LABEL
word  ~> =html#htags#paired('label','') 
abbr  ~> tlabel
menu  ~> <html>
--
